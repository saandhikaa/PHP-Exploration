Edit extra-keys
cd ~/.termux/
nano termux.properties
extra-keys=[[],[]]

Change font-family
mv font.ttf ~/.termux/

Create automatic commands run at start
nano ~/.bashrc or ~/.zshrc

switch to zsh terminal
pkg install git
pkg install zsh
sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"

change zsh theme to powerlevel10k
git clone --depth=1 https://github.com/romkatv/powerlevel10k.git $ZSH_CUSTOM/themes/powerlevel10k
nano ~/.zshrc
ZSH_THEME="powerlevel10k/powerlevel10k"
p10k configure

add zsh-autosuggestions
git clone https://github.com/zsh-users/zsh-autosuggestions $ZSH_CUSTOM/plugins/zsh-autosuggestions
nano ~/.zshrc
change to plugins=(git zsh-autosuggestions)
source ~/.zshrc

install php webserver
pkg update
pkg install php-apache
nano ~/../usr/etc/apache2/httpd.conf
comment out LoadModule mpm_prefork_module...
create:
    LoadModule php_module libexec/apache2/libphp.so
    <FilesMatch \.php$>
        SetHandler application/x-httpd-php
    </FilesMatch>
comment LoadModule mpm_worker_module...
comment out LoadModule rewrite_module...
comment out #ServerName www.example.com:8080
change DocumentRoot and Directory tag attribute to "/data/data/com.termux/files/home/storage/shared"
switch AllowOverride to All
add index.php at #directoryIndex
create phpinfo/index.php in root/ directory
    <?php phpinfo(); ?>
apachectl start
apachectl stop

install mariadb
pkg install mariadb
mysqld
open new session
    pkill mysqld

install phpmyadmin
download phpmyadmin from https://www.phpmyadmin.net
extract to root/ directory webserver
rename to phpmyadmin/
open termux
cd phpmyadmin/
cp config.sample.inc.php config.inc.php
nano config.inc.php
locate to $cfg['blowfish_secret']
get blowfish key from https://www.motorsportdiesel.com/tools/blowfish-salt/pma/
update $cfg['blowfish_secret'] = 'key'
change $cfg server host = '127.0.0.1'
change $cfg server AllowNoPassword = true
